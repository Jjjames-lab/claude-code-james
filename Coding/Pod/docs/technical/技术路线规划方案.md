### 2.2 数据库设计

```sql
-- 用户表
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT UNIQUE NOT NULL,
  name TEXT,
  avatar_url TEXT,
  subscription_plan TEXT DEFAULT 'free', -- free, pro, team
  quota_used INTEGER DEFAULT 0, -- 本月已使用次数
  quota_limit INTEGER DEFAULT 3, -- 本月限制次数
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- 播客表
CREATE TABLE podcasts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  xiaoyuzhou_id TEXT UNIQUE NOT NULL,
  title TEXT NOT NULL,
  podcast_name TEXT,
  audio_url TEXT NOT NULL,
  duration INTEGER, -- 秒
  cover_image TEXT,
  show_notes TEXT,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- 转录表
CREATE TABLE transcripts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  podcast_id UUID REFERENCES podcasts(id) ON DELETE CASCADE,
  data JSONB NOT NULL, -- 完整转录数据
  asr_engine TEXT NOT NULL, -- doubao, whisper
  word_count INTEGER,
  processing_time INTEGER, -- 秒
  created_at TIMESTAMP DEFAULT NOW()
);

-- 书签表 (Pins)
CREATE TABLE bookmarks (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  podcast_id UUID REFERENCES podcasts(id) ON DELETE CASCADE,
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  start_time DECIMAL(10,3) NOT NULL, -- 秒
  end_time DECIMAL(10,3) NOT NULL,
  text TEXT NOT NULL,
  note TEXT, -- 用户自定义笔记
  created_at TIMESTAMP DEFAULT NOW()
);

-- 知识笔记表 (差异化功能)
CREATE TABLE knowledge_notes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  podcast_id UUID REFERENCES podcasts(id) ON DELETE CASCADE,
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  title TEXT NOT NULL,
  content TEXT NOT NULL,
  tags TEXT[], -- 标签数组
  related_segments JSONB, -- 相关文本片段
  knowledge_graph JSONB, -- 知识图谱数据
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- 学习进度表
CREATE TABLE learning_progress (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  podcast_id UUID REFERENCES podcasts(id) ON DELETE CASCADE,
  last_position DECIMAL(10,3) DEFAULT 0, -- 最后播放位置
  completion_rate DECIMAL(5,2) DEFAULT 0, -- 完成百分比
  bookmark_count INTEGER DEFAULT 0,
  note_count INTEGER DEFAULT 0,
  studied_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(user_id, podcast_id)
);

-- 支付记录表
CREATE TABLE payments (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  stripe_payment_id TEXT UNIQUE,
  amount INTEGER NOT NULL, -- 分为单位
  currency TEXT DEFAULT 'cny',
  plan TEXT NOT NULL, -- pro, team
  status TEXT NOT NULL, -- pending, succeeded, failed
  created_at TIMESTAMP DEFAULT NOW()
);
```

### 2.3 API设计

```typescript
// 认证相关
POST /api/auth/signup        -- 用户注册
POST /api/auth/signin       -- 用户登录
POST /api/auth/signout      -- 用户登出
GET  /api/auth/me          -- 获取当前用户

// 播客相关
POST /api/podcasts/parse    -- 解析播客URL
GET  /api/podcasts          -- 获取用户播客列表
GET  /api/podcasts/:id     -- 获取单个播客
DELETE /api/podcasts/:id    -- 删除播客

// 转录相关
POST /api/transcripts       -- 创建转录任务
GET  /api/transcripts/:id   -- 获取转录结果
POST /api/transcripts/:id/retry -- 重试转录

// 书签相关
POST /api/bookmarks         -- 创建书签
GET  /api/bookmarks        -- 获取书签列表
PUT  /api/bookmarks/:id    -- 更新书签
DELETE /api/bookmarks/:id   -- 删除书签

// 知识笔记相关
POST /api/knowledge-notes   -- 创建知识笔记
GET  /api/knowledge-notes   -- 获取知识笔记列表
PUT  /api/knowledge-notes/:id -- 更新知识笔记
DELETE /api/knowledge-notes/:id -- 删除知识笔记

// 学习相关
GET  /api/learning/progress/:podcastId -- 获取学习进度
PUT  /api/learning/progress/:podcastId -- 更新学习进度

// 支付相关
POST /api/payments/create-checkout-session -- 创建支付会话
POST /api/payments/webhook              -- Stripe Webhook
GET  /api/payments/subscription        -- 获取订阅状态
```
